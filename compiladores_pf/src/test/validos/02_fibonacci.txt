// N-esimo termino de Fibonacci, version iterativa
// Cubre: while, multiples variables locales, asignaciones encadenadas

int n;
int resultado;

def int fib(int x){
    int a;
    int b;
    int tmp;
    int i;
    a = 0;
    b = 1;
    i = 0;
    while(i < x){
        tmp = a + b;
        a = b;
        b = tmp;
        i = i + 1;
    }
    return a;
}

def int main(){
    n = 10;
    resultado = fib(n);
    return 0;
}
